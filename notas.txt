-- Crear y Conectarse a la base de datos del proyecto
CREATE DATABASE agrigestor_db;
CREATE USER agrigestor_user WITH PASSWORD 'agrigestor_pass';
GRANT ALL PRIVILEGES ON DATABASE agrigestor_db TO agrigestor_user;

\c agrigestor_db

-- Otorgar permisos completos sobre el schema public
GRANT ALL ON SCHEMA public TO agrigestor_user;
GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA public TO agrigestor_user;
GRANT ALL PRIVILEGES ON ALL SEQUENCES IN SCHEMA public TO agrigestor_user;

-- Importante: permisos para tablas que se creen en el futuro
ALTER DEFAULT PRIVILEGES IN SCHEMA public 
    GRANT ALL ON TABLES TO agrigestor_user;
ALTER DEFAULT PRIVILEGES IN SCHEMA public 
    GRANT ALL ON SEQUENCES TO agrigestor_user;

-- Salir
\q
